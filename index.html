param( [string]$Version = "1.0.0.0", [switch]$InstallAfterBuild, [switch]$NoSign ) $ErrorActionPreference = "Stop" $PackageName = "TarekWasfy.BigTIFFViewer" $Publisher = "CN=85434AF5-74BD-4E8F-90F0-13F9EA1270DE" $PublisherDisplayName = "Tarek Wasfy" $DisplayName = "BigTIFF Viewer" $Description = "BigTIFF Viewer" $Architecture = "x64" $Root = Split-Path -Parent $MyInvocation.MyCommand.Path $ExeDir = Join-Path $Root "EXE_Hier_Rein" $IconDir = Join-Path $Root "Icon_Hier_Rein" $RuntimeDir = Join-Path $Root "Runtime_Hier_Rein" $OutRoot = Join-Path $Root "Output" $Stage = Join-Path $OutRoot "Package" $Assets = Join-Path $Stage "Assets" $UnsignedMsix = Join-Path $OutRoot "$PackageName`_$Version`_$Architecture`_unsigned.msix" $SignedMsix = Join-Path $OutRoot "$PackageName`_$Version`_$Architecture.msix" function Write-Step($Text) { Write-Host "" Write-Host "=== $Text ===" -ForegroundColor Cyan } function Find-Tool([string]$ToolName) { $kits = Join-Path ${env:ProgramFiles(x86)} "Windows Kits\10\bin" if (!(Test-Path $kits)) { throw "Windows SDK nicht gefunden. Bitte Windows 10/11 SDK installieren. Fehlender Ordner: $kits" } $candidates = Get-ChildItem $kits -Directory -ErrorAction SilentlyContinue | Sort-Object Name -Descending | ForEach-Object { @( (Join-Path $_.FullName "x64\$ToolName"), (Join-Path $_.FullName "x86\$ToolName") ) } | Where-Object { Test-Path $_ } $tool = $candidates | Select-Object -First 1 if (!$tool) { throw "$ToolName nicht gefunden. Bitte Windows SDK inklusive Packaging Tools installieren." } return $tool } function Get-InputExe { if (!(Test-Path $ExeDir)) { New-Item -ItemType Directory -Force $ExeDir | Out-Null } $preferred = Join-Path $ExeDir "BigTIFFViewer.exe" if (Test-Path $preferred) { return $preferred } $exe = Get-ChildItem $ExeDir -Filter *.exe -File -ErrorAction SilentlyContinue | Select-Object -First 1 if (!$exe) { throw "Keine EXE gefunden. Lege deine BigTIFFViewer.exe hier ab: $ExeDir" } return $exe.FullName } function Get-InputIcon { if (!(Test-Path $IconDir)) { New-Item -ItemType Directory -Force $IconDir | Out-Null } $icons = Get-ChildItem $IconDir -File -ErrorAction SilentlyContinue | Where-Object { $_.Extension.ToLowerInvariant() -in @(".png",".ico",".jpg",".jpeg",".bmp") } | Sort-Object @{Expression={ if ($_.Name -match "app|icon|logo") { 0 } else { 1 }}}, Name $icon = $icons | Select-Object -First 1 if (!$icon) { throw "Kein Icon gefunden. Lege eine .png, .ico, .jpg, .jpeg oder .bmp Datei hier ab: $IconDir" } return $icon.FullName } function Load-SourceImage([string]$Path) { Add-Type -AssemblyName System.Drawing $ext = [IO.Path]::GetExtension($Path).ToLowerInvariant() if ($ext -eq ".ico") { $ico = New-Object System.Drawing.Icon($Path) try { $bmp = $ico.ToBitmap() return New-Object System.Drawing.Bitmap($bmp) } finally { if ($bmp) { $bmp.Dispose() } $ico.Dispose() } } return [System.Drawing.Image]::FromFile($Path) } function Save-Asset([System.Drawing.Image]$Source, [string]$OutFile, [int]$W, [int]$H, [switch]$Crop) { Add-Type -AssemblyName System.Drawing $bmp = New-Object System.Drawing.Bitmap($W, $H, [System.Drawing.Imaging.PixelFormat]::Format32bppArgb) $g = [System.Drawing.Graphics]::FromImage($bmp) try { $g.Clear([System.Drawing.Color]::Transparent) $g.InterpolationMode = [System.Drawing.Drawing2D.InterpolationMode]::HighQualityBicubic $g.SmoothingMode = [System.Drawing.Drawing2D.SmoothingMode]::HighQuality $g.PixelOffsetMode = [System.Drawing.Drawing2D.PixelOffsetMode]::HighQuality $g.CompositingQuality = [System.Drawing.Drawing2D.CompositingQuality]::HighQuality $sx = [double]$W / [double]$Source.Width $sy = [double]$H / [double]$Source.Height if ($Crop) { $scale = [Math]::Max($sx, $sy) } else { $scale = [Math]::Min($sx, $sy) } $dw = [int][Math]::Round($Source.Width * $scale) $dh = [int][Math]::Round($Source.Height * $scale) $dx = [int][Math]::Round(($W - $dw) / 2.0) $dy = [int][Math]::Round(($H - $dh) / 2.0) $dest = New-Object System.Drawing.Rectangle($dx, $dy, $dw, $dh) $g.DrawImage($Source, $dest) $bmp.Save($OutFile, [System.Drawing.Imaging.ImageFormat]::Png) } finally { $g.Dispose() $bmp.Dispose() } } function Create-Assets([string]$IconPath) { Write-Step "Erzeuge MSIX Assets aus Icon" New-Item -ItemType Directory -Force $Assets | Out-Null $src = Load-SourceImage $IconPath try { Save-Asset $src (Join-Path $Assets "Square44x44Logo.png") 44 44 Save-Asset $src (Join-Path $Assets "Square71x71Logo.png") 71 71 Save-Asset $src (Join-Path $Assets "Square150x150Logo.png") 150 150 Save-Asset $src (Join-Path $Assets "Square310x310Logo.png") 310 310 Save-Asset $src (Join-Path $Assets "StoreLogo.png") 50 50 Save-Asset $src (Join-Path $Assets "Wide310x150Logo.png") 310 150 Save-Asset $src (Join-Path $Assets "SplashScreen.png") 620 300 foreach ($s in @(16,20,24,30,32,36,40,44,48,60,64,72,80,96,256)) { Save-Asset $src (Join-Path $Assets "Square44x44Logo.targetsize-$s.png") $s $s Save-Asset $src (Join-Path $Assets "Square44x44Logo.targetsize-$s`_altform-unplated.png") $s $s } } finally { $src.Dispose() } } function Write-Manifest { Write-Step "Schreibe AppxManifest.xml" $manifest = @" $DisplayName $PublisherDisplayName Assets\StoreLogo.png TIFF / BigTIFF Image .tif .tiff .btf "@ $manifest | Set-Content -Path (Join-Path $Stage "AppxManifest.xml") -Encoding UTF8 } function Prepare-Stage([string]$ExePath, [string]$IconPath) { Write-Step "Bereite Package-Ordner vor" if (Test-Path $Stage) { Remove-Item $Stage -Recurse -Force } New-Item -ItemType Directory -Force $Stage | Out-Null New-Item -ItemType Directory -Force $OutRoot | Out-Null Copy-Item $ExePath (Join-Path $Stage "BigTIFFViewer.exe") -Force $runtimeGdal = Join-Path $RuntimeDir "GDAL" if (Test-Path $runtimeGdal) { Write-Host "Kopiere GDAL Runtime..." New-Item -ItemType Directory -Force (Join-Path $Stage "runtimes") | Out-Null Copy-Item $runtimeGdal (Join-Path $Stage "runtimes") -Recurse -Force } else { Write-Host "Keine optionale GDAL Runtime gefunden: $runtimeGdal" } Create-Assets $IconPath Write-Manifest } function Create-TestCertIfNeeded { Write-Step "Prüfe Test-Zertifikat" $subject = $Publisher $cert = Get-ChildItem Cert:\CurrentUser\My | Where-Object { $_.Subject -eq $subject } | Sort-Object NotAfter -Descending | Select-Object -First 1 if (!$cert) { Write-Host "Erzeuge neues lokales Test-Zertifikat: $subject" $cert = New-SelfSignedCertificate ` -Type Custom ` -Subject $subject ` -KeyUsage DigitalSignature ` -FriendlyName "BigTIFF Viewer MSIX Test Certificate" ` -CertStoreLocation "Cert:\CurrentUser\My" ` -TextExtension @("2.5.29.37={text}1.3.6.1.5.5.7.3.3") } else { Write-Host "Vorhandenes Zertifikat gefunden: $($cert.Thumbprint)" } $trusted = Get-ChildItem Cert:\CurrentUser\TrustedPeople | Where-Object { $_.Thumbprint -eq $cert.Thumbprint } | Select-Object -First 1 if (!$trusted) { Write-Host "Importiere Test-Zertifikat nach CurrentUser\TrustedPeople..." $tmpCer = Join-Path $OutRoot "BigTIFFViewer_TestCert.cer" Export-Certificate -Cert $cert -FilePath $tmpCer | Out-Null Import-Certificate -FilePath $tmpCer -CertStoreLocation "Cert:\CurrentUser\TrustedPeople" | Out-Null } return $cert } function Build-MSIX { $makeappx = Find-Tool "makeappx.exe" Write-Step "Erzeuge MSIX" if (Test-Path $UnsignedMsix) { Remove-Item $UnsignedMsix -Force } if (Test-Path $SignedMsix) { Remove-Item $SignedMsix -Force } & $makeappx pack /d $Stage /p $UnsignedMsix /o if ($LASTEXITCODE -ne 0) { throw "makeappx fehlgeschlagen. Exit Code: $LASTEXITCODE" } if ($NoSign) { Write-Host "NoSign aktiv. Unsigned MSIX:" Write-Host $UnsignedMsix -ForegroundColor Yellow return $UnsignedMsix } $signtool = Find-Tool "signtool.exe" $cert = Create-TestCertIfNeeded Write-Step "Signiere MSIX" Copy-Item $UnsignedMsix $SignedMsix -Force & $signtool sign /fd SHA256 /sha1 $cert.Thumbprint $SignedMsix if ($LASTEXITCODE -ne 0) { throw "signtool fehlgeschlagen. Exit Code: $LASTEXITCODE" } return $SignedMsix } try { Write-Step "BigTIFF Viewer MSIX Builder" Write-Host "Root: $Root" $exe = Get-InputExe $icon = Get-InputIcon Write-Host "EXE: $exe" Write-Host "Icon: $icon" Prepare-Stage $exe $icon $package = Build-MSIX Write-Step "FERTIG" Write-Host "MSIX: $package" -ForegroundColor Green Write-Host "Output: $OutRoot" if ($InstallAfterBuild) { Write-Step "Installiere lokal" Add-AppxPackage -Path $package -ForceApplicationShutdown Write-Host "Installiert." -ForegroundColor Green } } catch { Write-Host "" Write-Host "FEHLER:" -ForegroundColor Red Write-Host $_.Exception.Message -ForegroundColor Red Write-Host "" Write-Host "Kontrolle:" -ForegroundColor Yellow Write-Host "1. EXE liegt in: $ExeDir" Write-Host "2. Icon liegt in: $IconDir" Write-Host "3. Windows SDK mit makeappx.exe und signtool.exe ist installiert." exit 1 }
